Step 1 - PYPG Update bash_profile

Edit ~/.bashrc and append the following to the end of the file.

Connect via ssh:

ssh {linuxattuneuser}@{attunenode}

Execute the following script: 

                
sed -i '/ATTUNE_PY_VER/d' ~/.bash_profile
sed -i '/ATTUNE_PG_VER/d' ~/.bash_profile
sed -i '/ATTUNE_CMAKE_VER/d' ~/.bash_profile
sed -i '/ATTUNE_TSDB_VER/d' ~/.bash_profile

cat <<"EOF">> ~/.bash_profile

##### SET THE ATTUNE ENVIRONMENT #####
# Setup the variables for PYTHON and POSTGRESQL
export ATTUNE_PY_VER="{attunePythonVer}"
export ATTUNE_PG_VER="{attunePostgresqlVer}"
export ATTUNE_CMAKE_VER="{attuneCmakeVer}"
export ATTUNE_TSDB_VER="{attuneTimescaledbVer}"

EOF

Step 4 - Install PostgreSQL

Step 4.1 - IPG Deploy PostgreSQL

Copy the File(s) Offline_PostgreSQL_Source.tar.bz2 to the target node.

Deploy the File(s) here:

/home/{linuxAttuneUser.user}

Step 4.2 - IPG Configure Build Install

Configure the build

Execute the following script: 

                
ATTUNE_PG_VER={attunePostgresqlVer}

SRC_DIR="$HOME/postgresql-${ATTUNE_PG_VER}"

# Remove the src dir and install file
rm -rf ${SRC_DIR} || true

tar xjf postgresql-${ATTUNE_PG_VER}.tar.bz2


cd ${SRC_DIR}

export CPPFLAGS=" -I`echo $HOME/opt/include/python*m` "
export LDFLAGS=" -L$HOME/opt/lib "

./configure \
      --disable-debug \
      --prefix=$HOME/opt \
      --enable-thread-safety \
      --with-openssl \
      --with-python


make -j4 > make.log

make install-world > make-install.log

# this is required for timescale to compile
cp ${SRC_DIR}/src/test/isolation/pg_isolation_regress ~/opt/bin

# Remove the src dir and install file
rm -rf ${SRC_DIR}*

Step 4.3 - IPG Init Cluster

Execute the following script: 

                
initdb --pgdata=$HOME/pgdata/15 --auth-local=trust  --auth-host=md5

Step 4.4 - IPG Tune postgresql.conf

Execute the following script: 

                
F="$HOME/pgdata/15/postgresql.conf"

sed -i 's/max_connections = 100/max_connections = 200/g' $F

# Print the changed line
grep max_connections $F

Step 4.5 - IPG Install Service File

Connect via ssh:

ssh {linuxrootuser}@{attunenode}

Execute the following script: 

                
F=/usr/lib/systemd/system/postgresql-15.service

cat <<"EOF" | sed "s,\$HOME,`echo ~{linuxAttuneUser.user}`,g" > $F
[Unit]
Description=PostgreSQL 15 database server
After=network-online.target

[Service]
Type=forking
User={linuxAttuneUser.user}
Group={linuxAttuneUser.user}

# Location of database directory
Environment=PGDATA=$HOME/pgdata/15

# Disable OOM kill on the postmaster
OOMScoreAdjust=-1000
Environment=PG_OOM_ADJUST_FILE=/proc/self/oom_score_adj
Environment=PG_OOM_ADJUST_VALUE=0

ExecStart=$HOME/opt/bin/pg_ctl -D ${PGDATA} start
ExecStop=$HOME/opt/bin/pg_ctl -D ${PGDATA} stop
ExecReload=/bin/kill -HUP $MAINPID
KillMode=mixed
KillSignal=SIGINT
 

# Do not set any timeout value, so that systemd will not kill postmaster
# during crash recovery.
TimeoutSec=0

[Install]
WantedBy=multi-user.target
EOF

systemctl daemon-reload

# Print files, and replace empty lines with a . so Attune shows them
systemctl cat postgresql-15 | sed 's/^$/./g'

Step 7 - Configure SELinux

Step 7.1 - Setup Attune SELinux

NOTE: These SELinux rules are tested for Field and Office only nodes.

Some more work is required for other Peeks services.

Connect via ssh:

ssh {linuxrootuser}@{attunenode}

Execute the following script: 

                
cat <<EOF > attune.te
module attune 1.0;

require {
        type unreserved_port_t;
        type user_home_t;
        type ptmx_t;
        type proc_t;
        type devpts_t;
        type groupadd_t;
        type init_t;
        type vmtools_unconfined_exec_t;
        class file { append create execute execute_no_trans ioctl map open read lock rename write };
        class lnk_file read;
        class process execmem;
        class tcp_socket name_connect;
        class chr_file { ioctl open read write };
        class filesystem getattr;
}

#============= init_t ==============

allow groupadd_t proc_t:filesystem getattr;
allow init_t devpts_t:chr_file open;
allow init_t ptmx_t:chr_file { ioctl open read write };
allow init_t self:process execmem;
allow init_t unreserved_port_t:tcp_socket name_connect;
allow init_t user_home_t:file { append create execute execute_no_trans ioctl map open read lock rename write };
allow init_t user_home_t:lnk_file read;
allow init_t vmtools_unconfined_exec_t:file { read open ioctl };
EOF

checkmodule -M -m -o attune.mod attune.te
semodule_package -o attune.pp -m attune.mod
semodule -i attune.pp

rm -f attune.{te,mod,pp}

Step 8.2 - SAC Create Skeleton config.cfg

Execute the following script: 

                
[ -d ~/attune.home ] || mkdir ~/attune.home

if ! grep -q "[disk_storage]" ~/attune.home/config.cfg
then

    cat <<"EOF" >> ~/attune.home/config.cfg
    
[disk_storage]
home_path = /home/attune/attune.home
archives_path = %(home_path)s/archives_contents
tmp_path = %(home_path)s/tmp
archives_unlocked_contents = %(home_path)s/archives_unlocked_contents


[alembic]
# path to migration scripts
script_location = /home/{linuxAttuneUser.user}/python/lib/python3.12/site-packages/attune/alembic/migration
sourceless = true
sqlalchemy.url = postgresql://{postgresqlAttuneUser.user}:{postgresqlAttuneUser.password}@localhost/attune

[user]
admin.pass = {attuneWebappUserAdmin.password}



[http]
https_port = 8001
http_redirect_from_http_port = 8000
http_redirect_to_https_port = 443


EOF

else
    echo "[disk_storage] is already written, skipping."
fi

echo "~/attune.home/config.cfg:"
cat ~/attune.home/config.cfg
