Step 1 - LVM Add PVs to VGs

Connect via ssh:

ssh {linuxrootuser}@{attunenode}

Execute the following script: 

                
mkvg() {
    VG=$1
    DEV=$2

    while ! test -e /dev/$DEV
    do
        echo "Waiting for kernel to detect new disk"
        sleep 1
    done
    
    parted -s /dev/$DEV mktable gpt
    parted -s /dev/$DEV mkpart primary xfs 1m 100%
    
    vgextend $VG /dev/${DEV}1
}

if [ -n "{lvmVgDisksToAdd}" ]
then
    for disk in {lvmVgDisksToAdd}
    do
        mkvg rootvg $disk
    done
else
    echo "SKIPPING: No disks configured"
fi

Step 3 - LVM Create Backup LV

Execute the following script: 

                
mklv() {
    DIR=$1
    NAME=$2
    SIZE=$3
    VG=$4
    
    mkdir -p $DIR
    lvcreate --size $SIZE --name $NAME $VG 
    mkfs.xfs /dev/mapper/$VG-$NAME
    
    cat >> /etc/fstab <<EOF
/dev/mapper/$VG-$NAME    $DIR                    xfs     defaults        0 0
EOF
    mount -a
}

mklv /backup backup 50g rootvg
